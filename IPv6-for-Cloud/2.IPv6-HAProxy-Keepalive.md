# Cấu hình dual stack IPv6 - IPv4 HAProxy + Keepalived
## 1. Cấu hình DNS server
- 1 domain sẽ trỏ về 1 IPv4 và 1 IPv6
- `/etc/bind/named.conf.local`

```
zone "bizfly.test"{
        type master;
        file "/etc/bind/db.bizfly.test";
};

zone "12.0.10.203.in-addr.arpa"{
        type master;
        file "/etc/bind/db.bizfly.test.reverse";
};
```

- `/etc/bind/db.bizfly.test`
```
$TTL    86400
@       IN      SOA     hn.bizfly.test. root.bizfly.test. (
                              1         ; Serial
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                          86400 )       ; Negative Cache TTL
;
@       IN      NS      hn.bizfly.test.
@       IN      A       172.16.1.6
hn      IN      A       203.10.0.12
www     IN      CNAME   hn
```

- `/etc/bind/db.bizfly.test.reverse`
```
$TTL    604800
@       IN      SOA     hn.bizfly.test. root.bizfly.test. (
                              2         ; Serial
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                         604800 )       ; Negative Cache TTL
;
@               IN      NS      hn.bizfly.test.
203.10.0.12     IN      PTR     hn.bizfly.test.
```

## 2. Cấu hình support
- `netplan`
```
network:
  version: 2
  ethernets:
          ens3:
            dhcp4: no
          ens4:
            dhcp4: no
          ens5:
            dhcp4: no
            addresses: [172.16.1.3/24]
  vlans:
       vlan234:
               id: 234
               link: ens4
               #               addresses: [203.10.0.11/24]
       vlan123:
               id: 123
               link: ens4
  bridges:
          br-vlan234:
                  interfaces: [vlan234]
                  addresses: [203.10.0.11/24]
                  nameservers:
                        addresses: [172.16.1.6]
                        search: [localdomain]
          br-vlan123:
                  interfaces: [vlan123]
                  addresses: [2405:f980:0:1::3/64]
                  nameservers:
                          addresses: [172.16.1.6]
                          search: [localdomain]
```

- `keepalived`
```
vrrp_instance WAN_v4 {
    state BACKUP
    interface br-vlan234
    virtual_router_id 50
    dont_track_primary
    priority 150
    preempt_delay 30
    garp_master_delay 1
    advert_int 2
    track_interface {
        br-vlan234
    }
    virtual_ipaddress {
    203.10.0.12 dev br-vlan234
    }
}

vrrp_instance WAN_v6 {
    state BACKUP
    interface br-vlan123
    virtual_router_id 50
    dont_track_primary
    priority 150
    preempt_delay 30
    garp_master_delay 1
    advert_int 2
    track_interface {
        br-vlan123
    }
    virtual_ipaddress {
    2405:f980:0:1::4 dev br-vlan123
    }
}
```

- `haproxy`
```
global
        log /dev/log    local0
        log /dev/log    local1 notice
        chroot /var/lib/haproxy
        stats socket /run/haproxy/admin.sock mode 660 level admin expose-fd listeners
        stats timeout 30s
        user haproxy
        group haproxy
        daemon

        # Default SSL material locations
        ca-base /etc/ssl/certs
        crt-base /etc/ssl/private

        # See: https://ssl-config.mozilla.org/#server=haproxy&server-version=2.0.3&config=intermediate
        ssl-default-bind-ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384
        ssl-default-bind-ciphersuites TLS_AES_128_GCM_SHA256:TLS_AES_256_GCM_SHA384:TLS_CHACHA20_POLY1305_SHA256
        ssl-default-bind-options ssl-min-ver TLSv1.2 no-tls-tickets

defaults
        log     global
        mode    http
        option  httplog
        option  dontlognull
        timeout connect 5000
        timeout client  50000
        timeout server  50000
        errorfile 400 /etc/haproxy/errors/400.http
        errorfile 403 /etc/haproxy/errors/403.http
        errorfile 408 /etc/haproxy/errors/408.http
        errorfile 500 /etc/haproxy/errors/500.http
        errorfile 502 /etc/haproxy/errors/502.http
        errorfile 503 /etc/haproxy/errors/503.http
        errorfile 504 /etc/haproxy/errors/504.http

frontend web-frontend
#        bind hn.bizfly.test:8080 
        bind 203.10.0.12:8080
        bind 2405:f980:0:1::4:8080
        mode http
        use_backend web-backend

backend web-backend
        balance roundrobin
        server support-1 203.10.0.10:80 check
        server support-2 203.10.0.11:80 check

```
